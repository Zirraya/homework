    \section{Префикс-функция}
    \subsection{Текст программы}
       \begin{verbatim}
        // Префикс функция
vector<int> Prefix_Function(const string& s) {
    int n = s.length();
    vector<int> pi(n, 0); //вектор для хранения значений префикс-функции

    for (int i = 1; i < n; ++i) {
        int j = pi[i - 1]; //длина предыдущего префикса

        //пока символы не совпадают и j больше 0, уменьшаем j
        while (j > 0 && s[i] != s[j]) {
            j = pi[j - 1];
        }

        //если символы совпадают, увеличиваем j
        if (s[i] == s[j]) {
            j++;
        }

        pi[i] = j; //записываем значение префикс-функции
    }

    return pi;
}
//

 \end{verbatim}
    \subsection{Анализ}

Алгоритм префикс-функции имеет временную сложность $O(n)$, где $n$ - длина строки.

\textbf{Доказательство:}

\begin{itemize}
    \item Внешний цикл: $n-1$ итераций
    \item Внутренний цикл \texttt{while}: кажется $O(n)$ в худшем случае, но амортизированно $O(1)$
    \item Ключевое наблюдение: $\pi[i] \leq \pi[i-1] + 1$
    \item Переменная $j$ (текущая длина префикса) увеличивается максимум на 1 за итерацию
    \item Каждое уменьшение $j$ в \texttt{while} соответствует предыдущему увеличению
    \item Суммарно количество уменьшений $\leq$ суммарному количеству увеличений $\leq n$
\end{itemize}

Итого: $O(n)$ операций.


\section{Z-функция}
\subsection{Текст программы}
\begin{verbatim}
    // Z функция
vector<int> Z_function(const std::string& s) {
    int n = s.length();
    std::vector<int> Z(n, 0);
    int L = 0, R = 0;

    for (int i = 1; i < n; ++i) {
        if (i > R) {
            L = R = i;
            while (R < n && s[R] == s[R - L]) {
                R++;
            }
            Z[i] = R - L;
            R--;
        }
        else {
            int k = i - L;
            if (Z[k] < R - i + 1) {
                Z[i] = Z[k];
            }
            else {
                L = i;
                while (R < n && s[R] == s[R - L]) {
                    R++;
                }
                Z[i] = R - L;
                R--;
            }
        }
    }
    return Z;
}
//
\end{verbatim}

\subsection{Анализ}
Алгоритм Z-функции имеет временную сложность $O(n)$.

\textbf{Доказательство:}
\begin{itemize}
    \item Внешний цикл: $n-1$ итераций
    \item Ключевое наблюдение: указатель $R$ (правая граница Z-блока) движется только вправо
    \item На каждой итерации $R$ увеличивается, причем $R \leq n$
    \item Суммарное движение $R$: $O(n)$
    \item Все остальные операции - константные
\end{itemize}

Итого: $O(n)$ операций.

    //
void KnutMorrisPratt(const string& text, const string& pattern) {
    vector<int> pi = Prefix_Function(pattern);
    int n = text.length();
    int m = pattern.length();
    int j = 0; // индекс для pattern

    for (int i = 0; i < n; i++) {
        while (j > 0 && text[i] != pattern[j]) {
            j = pi[j - 1]; // откат к предыдущему префиксу
        }
        if (text[i] == pattern[j]) {
            j++;
        }
        if (j == m) { // найдено совпадение
            cout << (i - m + 1) << endl;
            j = pi[j - 1]; // продолжаем поиск
        }
    }
}
//
\end{verbatim}

\subsection{Анализ}

Алгоритм КМП имеет временную сложность $O(n + m)$, где:
\begin{itemize}
    \item $n$ - длина текста
    \item $m$ - длина образца
\end{itemize}

\textbf{Доказательство:}
\begin{itemize}
    \item Построение префикс-функции: $O(m)$
    \item Поиск в тексте: внешний цикл - $n$ итераций
    \item Амортизированный анализ: переменная $j$ увеличивается максимум на 1 за итерацию
    \item Суммарное количество откатов $j$ в \texttt{while} есть $O(n)$
    \item Все операции внутри циклов - константные
\end{itemize}

Итого: $O(n + m)$.
