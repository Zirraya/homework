// Функция для создания таблицы смещений по символам
vector<int> CreateBadCharTable(const string& pattern) {
    const int ALPHABET_SIZE = 256;
    vector<int> badCharTable(ALPHABET_SIZE, -1);

    for (int i = 0; i < pattern.size(); ++i) {
        badCharTable[static_cast<int>(pattern[i])] = i;
    }

    return badCharTable;
}
//

// Функция для поиска подстроки с использованием алгоритма Бойера — Мура
void BoyerMoore(const string& text, const string& pattern) {
    int m = pattern.size();
    int n = text.size();
    vector<int> badCharTable = CreateBadCharTable(pattern);

    int s = 0;
    bool first = true; // Флаг для определения первого вывода

    while (s <= n - m) {
        int j = m - 1;

        while (j >= 0 && pattern[j] == text[s + j]) {
            j--;
        }

        if (j < 0) {
            if (!first) {
                cout << " "; // Пробел между числами
            }
            cout << s;
            first = false;
            s += (s + m < n) ? m - badCharTable[static_cast<int>(text[s + m])] : 1;
        }
        else {
            s += max(1, j - badCharTable[static_cast<int>(text[s + j])]);
        }
    }

    // Если совпадений не найдено, вывод сообщение
    if (first) {
        cout << "Совпадений не найдено";
    }
}

\section{Анализ}

\textbf{Временная сложность алгоритма Бойера-Мура:}

\begin{itemize}
    \item \textbf{Предобработка:} $O(m + |\Sigma|)$, где $|\Sigma|$ - размер алфавита
    \item \textbf{Худший случай:} $O(n \cdot m)$
    \item \textbf{Лучший случай:} $O(n/m)$  
    \item \textbf{Средний случай на случайных текстах:} $O(n)$
\end{itemize}

\textbf{Обоснование:}

\begin{itemize}
    \item \textbf{Предобработка:} Построение таблицы "плохих символов" требует $O(|\Sigma|)$ памяти и $O(m)$ времени
    
    \item \textbf{Худший случай:} Возникает, когда шаблон и текст имеют периодическую структуру (например, \texttt{text = "AAA...AAA"}, \texttt{pattern = "AAA"}). На каждой из $O(n)$ позиций выполняется почти полное сравнение шаблона за $O(m)$
    
    \item \textbf{Лучший случай:} Когда первый же символ текста не встречается в шаблоне, алгоритм сдвигается на $m$ позиций за $O(1)$. Количество сдвигов: $O(n/m)$
    
     К примеру.  У нас есть текст: $"XXXXXXXXXXXX" (n=12,$ все символы X).

  А в шаблоне  $ABC: "ABC"$ $(m=3).$ Символ X не встречается в "ABC".
    

    
    \item \textbf{Средний случай:} На случайных текстах алгоритм демонстрирует сублинейное поведение. Эвристика "плохого символа" позволяет делать сдвиги в среднем близкие к $m$, что дает $O(n)$ сравнений
\end{itemize}

\textbf{Практическая эффективность:}

Несмотря на квадратичную худшую сложность, алгоритм Бойера-Мура часто оказывается быстрее алгоритмов с гарантированной линейной сложностью (как КМП) благодаря:
\begin{itemize}
    \item Большим сдвигам шаблона
    \item Проверке символов справа налево
    \item Эффективной работе с естественными языками
\end{itemize}
